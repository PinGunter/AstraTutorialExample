174void Astra::FreeCameraController::move(bool front, bool back, bool right, bool left, bool up, bool down, float speed)
175{
176 glm::vec3 frontBack = _camera.centre - _camera.eye;
177 float length = glm::length(frontBack);
178 glm::vec3 direction = glm::normalize(_camera.centre - _camera.eye);
179 glm::vec3 rightLeft = glm::normalize(glm::cross(frontBack, _camera.up));
180
181 float zSpeed = 0, xSpeed = 0, ySpeed = 0;
182
183 /*
184 The following sequence of ifs could be refactored to this:
185
186 xSpeed = right ? speed : left ? -speed : 0;
187 ySpeed = up ? speed : down ? -speed : 0;
188 zSpeed = front ? speed : back ? -speed : 0;
189
190 However, this sequence is left instead of nesting ternary operators
191 so that it is easier to read
192 */
193 if (front)
194 {
195 zSpeed += speed;
196 }
197 if (back)
198 {
199 zSpeed -= speed;
200 }
201 if (right)
202 {
203 xSpeed += speed;
204 }
205 if (left)
206 {
207 xSpeed -= speed;
208 }
209 if (up)
210 {
211 ySpeed += speed;
212 }
213 if (down)
214 {
215 ySpeed -= speed;
216 }
217
218 _camera.eye = _camera.eye + direction * zSpeed + rightLeft * xSpeed + _camera.up * ySpeed;
219 _camera.centre = _camera.eye + direction * length;

222void Astra::FreeCameraController::rotate(float dx, float dy)
223{
224
225 // Get the length of sight
226 glm::vec3 eyeToCenter(_camera.centre - _camera.eye);
227 float radius = glm::length(eyeToCenter);
228 eyeToCenter = glm::normalize(eyeToCenter);
229 glm::vec3 direction = eyeToCenter;
230
231 // Find the rotation around the UP axis (Y)
232 glm::mat4 rot_y = glm::rotate(glm::mat4(1), -dx, _camera.up);
233
234 // Apply the (Y) rotation to the eye-center vector
235 eyeToCenter = rot_y * glm::vec4(eyeToCenter, 0);
236
237 // Find the rotation around the X vector: cross between eye-center and up (X)
238 glm::vec3 axe_x = glm::normalize(glm::cross(_camera.up, direction));
239 glm::mat4 rot_x = glm::rotate(glm::mat4(1), dy, axe_x);
240
241 // Apply the (X) rotation to the eye-center vector
242 glm::vec3 vect_rot = rot_x * glm::vec4(eyeToCenter, 0);
243
244 if (glm::sign(vect_rot.x) == glm::sign(eyeToCenter.x))
245 eyeToCenter = vect_rot;
246
247 // Make the vector as long as it was originally
248 eyeToCenter *= radius;
249
250 // Finding the new position
251 glm::vec3 newCentre = eyeToCenter + _camera.eye;
252
253 _camera.centre = newCentre;
254}
