103void Astra::Renderer::createSwapchain(const VkSurfaceKHR &surface, uint32_t width, uint32_t height, VkFormat colorFormat, VkFormat depthFormat)
104{
105 _size = VkExtent2D{width, height};
106 _colorFormat = colorFormat;
107 _depthFormat = depthFormat;
108 const auto &device = AstraDevice.getVkDevice();
109 const auto &physicalDevice = AstraDevice.getPhysicalDevice();
110 const auto &queue = AstraDevice.getQueue();
111 const auto &graphicsQueueIndex = AstraDevice.getGraphicsQueueIndex();
112 const auto &commandPool = AstraDevice.getCommandPool();
113
114 // Find the most suitable depth format
115 if (_depthFormat == VK_FORMAT_UNDEFINED)
116 {
117 auto feature = VK_FORMAT_FEATURE_DEPTH_STENCIL_ATTACHMENT_BIT;
118 for (const auto &f : {VK_FORMAT_D24_UNORM_S8_UINT, VK_FORMAT_D32_SFLOAT_S8_UINT, VK_FORMAT_D16_UNORM_S8_UINT})
119 {
120 VkFormatProperties formatProp{VK_STRUCTURE_TYPE_FORMAT_PROPERTIES_2};
121 vkGetPhysicalDeviceFormatProperties(physicalDevice, f, &formatProp);
122 if ((formatProp.optimalTilingFeatures & feature) == feature)
123 {
124 _depthFormat = f;
125 break;
126 }
127 }
128 }
129
130 _swapchain.init(device, physicalDevice, queue, graphicsQueueIndex, surface, static_cast<VkFormat>(colorFormat));
131 _size = _swapchain.update(_size.width, _size.height, false);
132 _colorFormat = static_cast<VkFormat>(_swapchain.getFormat());
133
134 // Create Synchronization Primitives
135 _fences.resize(_swapchain.getImageCount());
136 for (auto &fence : _fences)
137 {
138 VkFenceCreateInfo fenceCreateInfo{VK_STRUCTURE_TYPE_FENCE_CREATE_INFO};
139 fenceCreateInfo.flags = VK_FENCE_CREATE_SIGNALED_BIT;
140 vkCreateFence(device, &fenceCreateInfo, nullptr, &fence);
141 }
142
143 /*
144 * We use the CommandList type in the class attribute
145 * However, vulkan expects a VkCommandBuffer array
146 * we have a local array of VkCommandBuffer for allocation
147 * the proceed to convert it to the CommandList class
148 */
149 std::vector<VkCommandBuffer> cmdBufs;
150 // Command buffers store a reference to the frame buffer inside their render pass info
151 // so for static usage without having to rebuild them each frame, we use one per frame buffer
152 VkCommandBufferAllocateInfo allocateInfo{VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO};
153 allocateInfo.commandPool = commandPool;
154 allocateInfo.commandBufferCount = _swapchain.getImageCount();
155 allocateInfo.level = VK_COMMAND_BUFFER_LEVEL_PRIMARY;
156 cmdBufs.resize(_swapchain.getImageCount());
157 vkAllocateCommandBuffers(device, &allocateInfo, cmdBufs.data());
158 for (auto &cmdBuf : cmdBufs)
159 {
160 _commandLists.push_back({cmdBuf});
161 }
162
163 auto cmdList = Astra::CommandList::createTmpCmdList();
164 _swapchain.cmdUpdateBarriers(cmdList.getCommandBuffer());
165 cmdList.submitTmpCmdList();

168void Astra::Renderer::createOffscreenRender(nvvk::ResourceAllocatorDma &alloc)
169{
170 const auto &device = AstraDevice.getVkDevice();

172 alloc.destroy(_offscreenDepth);

174 // Creating the color image
175 {
176 auto colorCreateInfo = nvvk::makeImage2DCreateInfo(_size, _offscreenColorFormat,
177 VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT | VK_IMAGE_USAGE_SAMPLED_BIT | VK_IMAGE_USAGE_STORAGE_BIT);
178

181 VkSamplerCreateInfo sampler{VK_STRUCTURE_TYPE_SAMPLER_CREATE_INFO};
182 _offscreenColor = alloc.createTexture(image, ivInfo, sampler);
183 _offscreenColor.descriptor.imageLayout = VK_IMAGE_LAYOUT_GENERAL;
184 }
185
186 // Creating the depth buffer
187 auto depthCreateInfo = nvvk::makeImage2DCreateInfo(_size, _offscreenDepthFormat, VK_IMAGE_USAGE_DEPTH_STENCIL_ATTACHMENT_BIT);
188 {
189 nvvk::Image image = alloc.createImage(depthCreateInfo);

191 VkImageViewCreateInfo depthStencilView{VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO};
192 depthStencilView.viewType = VK_IMAGE_VIEW_TYPE_2D;
193 depthStencilView.format = _offscreenDepthFormat;
194 depthStencilView.subresourceRange = {VK_IMAGE_ASPECT_DEPTH_BIT, 0, 1, 0, 1};
195 depthStencilView.image = image.image;
196
197 _offscreenDepth = alloc.createTexture(image, depthStencilView);
198 }
199
200 // Setting the image layout for both color and depth
201 {
202 nvvk::CommandPool genCmdBuf(device, AstraDevice.getGraphicsQueueIndex());
203 auto cmdBuf = genCmdBuf.createCommandBuffer();

206 VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL, VK_IMAGE_ASPECT_DEPTH_BIT);

209 }
210
211 // Creating a renderpass for the offscreen
212 if (!_offscreenRenderPass)
213 {
214 _offscreenRenderPass = nvvk::createRenderPass(device, {_offscreenColorFormat}, _offscreenDepthFormat, 1, true,

219 std::vector<VkImageView> attachments = {_offscreenColor.descriptor.imageView, _offscreenDepth.descriptor.imageView};

221 vkDestroyFramebuffer(device, _offscreenFb, nullptr);
222 VkFramebufferCreateInfo info{VK_STRUCTURE_TYPE_FRAMEBUFFER_CREATE_INFO};
223 info.renderPass = _offscreenRenderPass;

460void Astra::Renderer::endFrame(const CommandList &cmdList)
461{
462 cmdList.end();
463 uint32_t imageIndex = _swapchain.getActiveImageIndex();
464 vkResetFences(AstraDevice.getVkDevice(), 1, &_fences[imageIndex]);
465
466 const uint32_t deviceMask = 0b0000'0001;
467 const std::array<uint32_t, 2> deviceIndex = {0, 1};
468
469 VkDeviceGroupSubmitInfo deviceGroupSubmitInfo{VK_STRUCTURE_TYPE_DEVICE_GROUP_SUBMIT_INFO_KHR};
470 deviceGroupSubmitInfo.waitSemaphoreCount = 1;
471 deviceGroupSubmitInfo.commandBufferCount = 1;
472 deviceGroupSubmitInfo.pCommandBufferDeviceMasks = &deviceMask;
473 deviceGroupSubmitInfo.signalSemaphoreCount = 1;
474 deviceGroupSubmitInfo.pSignalSemaphoreDeviceIndices = deviceIndex.data();
475 deviceGroupSubmitInfo.pWaitSemaphoreDeviceIndices = deviceIndex.data();
476
477 VkSemaphore semaphoreRead = _swapchain.getActiveReadSemaphore();
478 VkSemaphore semaphoreWrite = _swapchain.getActiveWrittenSemaphore();
479
480 // Pipeline stage at which the queue submission will wait (via pWaitSemaphores)
481 const VkPipelineStageFlags waitStageMask = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
482 std::vector<VkCommandBuffer> commandBuffers(_commandLists.size());
483 for (int i = 0; i < commandBuffers.size(); i++)
484 {
485 commandBuffers[i] = _commandLists[i].getCommandBuffer();
486 }
487 // The submit info structure specifies a command buffer queue submission batch
488 VkSubmitInfo submitInfo{VK_STRUCTURE_TYPE_SUBMIT_INFO};
489 submitInfo.pWaitDstStageMask = &waitStageMask; // Pointer to the list of pipeline stages that the semaphore waits will occur at
490 submitInfo.pWaitSemaphores = &semaphoreRead; // Semaphore(s) to wait upon before the submitted command buffer starts executing
491 submitInfo.waitSemaphoreCount = 1; // One wait semaphore
492 submitInfo.pSignalSemaphores = &semaphoreWrite; // Semaphore(s) to be signaled when command buffers have completed
493 submitInfo.signalSemaphoreCount = 1; // One signal semaphore
494 submitInfo.pCommandBuffers = &commandBuffers[imageIndex]; // Command buffers(s) to execute in this batch (submission)
495 submitInfo.commandBufferCount = 1; // One command buffer
496 submitInfo.pNext = &deviceGroupSubmitInfo;
497
498 // Submit to the graphics queue passing a wait fence
499 vkQueueSubmit(AstraDevice.getQueue(), 1, &submitInfo, _fences[imageIndex]);
500
501 // Presenting frame
502 _swapchain.present(AstraDevice.getQueue());
503}
