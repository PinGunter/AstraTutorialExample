180void Astra::RayTracingPipeline::createSBT(nvvk::ResourceAllocatorDma &alloc, const VkPhysicalDeviceRayTracingPipelinePropertiesKHR &rtProperties)
181{
182 uint32_t missCount{2};
183 uint32_t hitCount{1};
184 auto handleCount = 1 + missCount + hitCount;
185 uint32_t handleSize = rtProperties.shaderGroupHandleSize;
186
187 // the sbt buffer needs to have starting groups to be aligned and handles in the group to be aligned
188 uint32_t handleSizeAligned = nvh::align_up(handleSize, rtProperties.shaderGroupHandleAlignment);
189
190 _rgenRegion.stride = nvh::align_up(handleSizeAligned, rtProperties.shaderGroupBaseAlignment);
191 _rgenRegion.size = _rgenRegion.stride; // raygen size and stride have to be the same
192 _missRegion.stride = handleSizeAligned;
193 _missRegion.size = nvh::align_up(missCount * handleSizeAligned, rtProperties.shaderGroupBaseAlignment);
194 _hitRegion.stride = handleSizeAligned;
195 _hitRegion.size = nvh::align_up(hitCount * handleSizeAligned, rtProperties.shaderGroupBaseAlignment);
196
197 // get the shader group handles
198 uint32_t dataSize = handleCount * handleSize;
199 std::vector<uint8_t> handles(dataSize);
200 auto result = vkGetRayTracingShaderGroupHandlesKHR(AstraDevice.getVkDevice(), _pipeline, 0, handleCount, dataSize, handles.data());
201 assert(result == VK_SUCCESS);
202
203 // allocate buffer for storing the sbt
204 VkDeviceSize sbtSize = _rgenRegion.size + _missRegion.size + _hitRegion.size;
205 _rtSBTBuffer = alloc.createBuffer(
206 sbtSize,
207 VK_BUFFER_USAGE_TRANSFER_SRC_BIT | VK_BUFFER_USAGE_SHADER_DEVICE_ADDRESS_BIT | VK_BUFFER_USAGE_SHADER_BINDING_TABLE_BIT_KHR,
208 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT);
209
210 // find the sbt address of each group
211 VkBufferDeviceAddressInfo info{VK_STRUCTURE_TYPE_BUFFER_DEVICE_ADDRESS_INFO, nullptr, _rtSBTBuffer.buffer};
212 VkDeviceAddress sbtAddress = vkGetBufferDeviceAddress(AstraDevice.getVkDevice(), &info);
213 _rgenRegion.deviceAddress = sbtAddress;
214 _missRegion.deviceAddress = sbtAddress + _rgenRegion.size;
215 _hitRegion.deviceAddress = sbtAddress + _rgenRegion.size + _missRegion.size;
216
217 // helper to retrieve the handle data
218 auto getHandle = [&](int i)
219 { return handles.data() + i * handleSize; };
220
221 auto *pSBTBuffer = reinterpret_cast<uint8_t *>(alloc.map(_rtSBTBuffer));
222 uint8_t *pData{nullptr};
223 uint32_t handleIdx{0};
224
225 // raygen. Just copy handle
226 pData = pSBTBuffer;
227 memcpy(pData, getHandle(handleIdx++), handleSize);
228
229 // miss
230 pData = pSBTBuffer + _rgenRegion.size;
231 for (uint32_t c = 0; c < missCount; c++)
232 {
233 memcpy(pData, getHandle(handleIdx++), handleSize);
234 pData += _missRegion.stride;
235 }
236
237 // hit
238 pData = pSBTBuffer + _rgenRegion.size + _missRegion.size;
239 for (uint32_t c = 0; c < hitCount; c++)
240 {
241 memcpy(pData, getHandle(handleIdx++), handleSize);
242 pData += _hitRegion.stride;
243 }
244
245 alloc.unmap(_rtSBTBuffer);
246 alloc.finalizeAndReleaseStaging();
247}
