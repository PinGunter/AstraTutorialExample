268 nvvk::RaytracingBuilderKHR::BlasInput Device::objectToVkGeometry(const Astra::Mesh &model)
269 {
270 size_t nbIndices = model.indices.size();
271 size_t nbVertices = model.vertices.size();
272 // BLAS builder requires raw device addresses.
273 uint32_t maxPrimitiveCount = nbIndices / 3;
274
275 // Describe buffer as array of VertexObj.
276 VkAccelerationStructureGeometryTrianglesDataKHR triangles{VK_STRUCTURE_TYPE_ACCELERATION_STRUCTURE_GEOMETRY_TRIANGLES_DATA_KHR};
277 triangles.vertexFormat = VK_FORMAT_R32G32B32_SFLOAT; // vec3 vertex position data.
278 triangles.vertexData.deviceAddress = model.descriptor.vertexAddress;
279 triangles.vertexStride = sizeof(Vertex);
280 // Describe index data (32-bit unsigned int)
281 triangles.indexType = VK_INDEX_TYPE_UINT32;
282 triangles.indexData.deviceAddress = model.descriptor.indexAddress;
283 // Indicate identity transform by setting transformData to null device pointer.
284 // triangles.transformData = {};
285 triangles.maxVertex = nbVertices - 1;
286
287 // Identify the above data as containing opaque triangles.
288 VkAccelerationStructureGeometryKHR asGeom{VK_STRUCTURE_TYPE_ACCELERATION_STRUCTURE_GEOMETRY_KHR};
289 asGeom.geometryType = VK_GEOMETRY_TYPE_TRIANGLES_KHR;
290 asGeom.flags = VK_GEOMETRY_OPAQUE_BIT_KHR;
291 asGeom.geometry.triangles = triangles;
292
293 // The entire array will be used to build the BLAS.
294 VkAccelerationStructureBuildRangeInfoKHR offset{};
295 offset.firstVertex = 0;
296 offset.primitiveCount = maxPrimitiveCount;
297 offset.primitiveOffset = 0;
298 offset.transformOffset = 0;
299
300 // Our blas is made from only one geometry, but could be made of many geometries
301 nvvk::RaytracingBuilderKHR::BlasInput input;
302 input.asGeometry.emplace_back(asGeom);
303 input.asBuildOffsetInfo.emplace_back(offset);
304
305 return input;
306 }

318 const auto &cmdBuf = cmdList.getCommandBuffer();
319
320 // If no textures are present, create a dummy one to accommodate the pipeline layout
321 if (new_textures.empty() && textures.empty())
322 {
323 nvvk::Texture texture;
324
325 std::array<uint8_t, 4> color{255u, 255u, 255u, 255u};
326 VkDeviceSize bufferSize = sizeof(color);

328 auto imageCreateInfo = nvvk::makeImage2DCreateInfo(imgSize, format);
329
330 // Creating the dummy texture
331 nvvk::Image image = alloc.createImage(cmdBuf, bufferSize, color.data(), imageCreateInfo);
332 VkImageViewCreateInfo ivInfo = nvvk::makeImageViewCreateInfo(image.image, imageCreateInfo);
333 texture = alloc.createTexture(image, ivInfo, samplerCreateInfo);
334
335 // The image format must be in VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL
336 nvvk::cmdBarrierImageLayout(cmdBuf, texture.image, VK_IMAGE_LAYOUT_UNDEFINED, VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL);
337 textures.push_back(texture);
338 }
339 else
340 {
341 // Uploading all images
342 for (const auto &texture : new_textures)
343 {
344 std::vector<std::string> defaultSearchPaths = {
345 NVPSystem::exePath() + PROJECT_RELDIRECTORY,
346 NVPSystem::exePath() + PROJECT_RELDIRECTORY "..",
347 std::string(PROJECT_NAME),
348 };
349
350 std::stringstream o;
351 int texWidth, texHeight, texChannels;
352 o << "media/textures/" << texture;
353 std::string txtFile = nvh::findFile(o.str(), defaultSearchPaths, true);
354
355 stbi_uc *stbi_pixels = stbi_load(txtFile.c_str(), &texWidth, &texHeight, &texChannels, STBI_rgb_alpha);
356
357 std::array<stbi_uc, 4> color{255u, 0u, 255u, 255u};
358
359 stbi_uc *pixels = stbi_pixels;
360 // Handle failure
361 if (!stbi_pixels)

363 texWidth = texHeight = 1;
364 texChannels = 4;
365 pixels = reinterpret_cast<stbi_uc *>(color.data());
366 }
367
368 VkDeviceSize bufferSize = static_cast<uint64_t>(texWidth) * texHeight * sizeof(uint8_t) * 4;
369 auto imgSize = VkExtent2D{(uint32_t)texWidth, (uint32_t)texHeight};
370 auto imageCreateInfo = nvvk::makeImage2DCreateInfo(imgSize, format, VK_IMAGE_USAGE_SAMPLED_BIT, true);
371
372 {
373 nvvk::Image image = alloc.createImage(cmdBuf, bufferSize, pixels, imageCreateInfo);
374 nvvk::cmdGenerateMipmaps(cmdBuf, image.image, format, imgSize, imageCreateInfo.mipLevels);
375 VkImageViewCreateInfo ivInfo = nvvk::makeImageViewCreateInfo(image.image, imageCreateInfo);
376 nvvk::Texture texture = alloc.createTexture(image, ivInfo, samplerCreateInfo);
377
378 textures.push_back(texture);
379 }
380
381 stbi_image_free(stbi_pixels);
382 }
