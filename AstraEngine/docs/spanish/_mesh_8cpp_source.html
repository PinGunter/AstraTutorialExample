94{
95 tinyobj::ObjReader reader;
96 reader.ParseFromFile(path);
97 if (!reader.Valid())
98 {
99 Astra::Log("Error reading obj file: " + path + ", error: " + reader.Error(), ERR);
100 assert(reader.Valid());
101 }
102
103 if (!reader.Warning().empty())
104 {
105 Astra::Log("Error reading obj file: " + path + ", error: " + reader.Warning(), WARNING);
106 }
107
108 // Collecting the material in the scene
109 for (const auto &material : reader.GetMaterials())
110 {
111 WaveFrontMaterial m;
112 m.ambient = glm::vec3(material.ambient[0], material.ambient[1], material.ambient[2]);
113 m.diffuse = glm::vec3(material.diffuse[0], material.diffuse[1], material.diffuse[2]);
114 m.specular = glm::vec3(material.specular[0], material.specular[1], material.specular[2]);
115 m.emission = glm::vec3(material.emission[0], material.emission[1], material.emission[2]);
116 m.transmittance = glm::vec3(material.transmittance[0], material.transmittance[1], material.transmittance[2]);
117 m.dissolve = material.dissolve;
118 m.ior = material.ior;
119 m.shininess = material.shininess;
120 m.illum = material.illum;
121 if (!material.diffuse_texname.empty())
122 {
123 textures.push_back(material.diffuse_texname);
124 m.textureId = static_cast<int>(textures.size()) - 1;
125 }
126
127 materials.emplace_back(m);
128 }
129
130 // If there were none, add a default
131 if (materials.empty())
132 {
133 materials.push_back(WaveFrontMaterial());
134 }
135
136 const tinyobj::attrib_t &attrib = reader.GetAttrib();
137
138 for (const auto &shape : reader.GetShapes())
139 {
140 vertices.reserve(shape.mesh.indices.size() + vertices.size());
141 indices.reserve(shape.mesh.indices.size() + indices.size());
142 materialIndices.insert(materialIndices.end(), shape.mesh.material_ids.begin(), shape.mesh.material_ids.end());
143
144 for (const auto &index : shape.mesh.indices)
145 {
146 Vertex vertex = {};
147 const float *vp = &attrib.vertices[3 * index.vertex_index];
148 vertex.pos = {*(vp + 0), *(vp + 1), *(vp + 2)};
149
150 if (!attrib.normals.empty() && index.normal_index >= 0)
151 {
152 const float *np = &attrib.normals[3 * index.normal_index];
153 vertex.nrm = {*(np + 0), *(np + 1), *(np + 2)};
154 }
155
156 if (!attrib.texcoords.empty() && index.texcoord_index >= 0)
157 {
158 const float *tp = &attrib.texcoords[2 * index.texcoord_index + 0];
159 vertex.texCoord = {*tp, 1.0f - *(tp + 1)};
160 }
161
162 if (!attrib.colors.empty())
163 {
164 const float *vc = &attrib.colors[3 * index.vertex_index];
165 vertex.color = {*(vc + 0), *(vc + 1), *(vc + 2)};
166 }
167
168 vertices.push_back(vertex);
169 indices.push_back(static_cast<int>(indices.size()));
170 }
171 }
172
173 // Fixing material indices
174 for (auto &mi : materialIndices)
175 {
176 if (mi < 0 || mi > materials.size())
177 mi = 0;
178 }
179
180 // Compute normal when no normal were provided.
181 if (attrib.normals.empty())
182 {
183 for (size_t i = 0; i < indices.size(); i += 3)
184 {
185 Vertex &v0 = vertices[indices[i + 0]];
186 Vertex &v1 = vertices[indices[i + 1]];
187 Vertex &v2 = vertices[indices[i + 2]];
188
189 glm::vec3 n = glm::normalize(glm::cross((v1.pos - v0.pos), (v2.pos - v0.pos)));
190 v0.nrm = n;
191 v1.nrm = n;
192 v2.nrm = n;
193 }
194 }
